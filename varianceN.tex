\label{chap:varianceN}

\makeoproblem{Variance of \(N\) in open system}{pr:varianceN:1}{
\citep{jackson2000equilibrium} pr 3.14, \citep{kittel1980thermal} pr 5.10
}{
Show that for an open system
\begin{dmath}\label{eqn:varianceN:20}
\var(N) = \inv{\beta} \PDc{\mu}{\overbar{N}}{V, T}.
\end{dmath}
} % makeoproblem

\makeanswer{pr:varianceN:1}{

In terms of the grand partition function, we find the (scaled) average number of particles
\begin{dmath}\label{eqn:varianceN:40}
\PD{\mu}{} \ln \ZG
=
\PD{\mu}{} \ln \sum_{r,s} e^{\beta \mu N_r - \beta E_s}
=
\inv{\ZG} \sum_{r,s} \beta N_r e^{\beta \mu N_r - \beta E_s}
=
\beta \overbar{N}.
\end{dmath}

Our second derivative provides us a scaled variance
\begin{dmath}\label{eqn:varianceN:60}
\PDSq{\mu}{} \ln \ZG
=
\PD{\mu}{}
\lr{
\inv{\ZG} \sum_{r,s} \beta N_r e^{\beta \mu N_r - \beta E_s}
}
=
\inv{\ZG} \sum_{r,s} (\beta N_r)^2 e^{\beta \mu N_r - \beta E_s}
-
\inv{\ZG^2}
\lr{\sum_{r,s} \beta N_r e^{\beta \mu N_r - \beta E_s}}^2
= \beta^2
\lr{
\overbar{N^2} - {\overbar{N}}^2
}.
\end{dmath}

Together this gives us the desired result
\begin{dmath}\label{eqn:varianceN:80}
\var(N)
= \inv{\beta^2}
\PD{\mu}{} \lr{ \beta \overbar{N} }
= \inv{\beta}
\PD{\mu}{\overbar{N}}.
\end{dmath}

} % makeanswer
