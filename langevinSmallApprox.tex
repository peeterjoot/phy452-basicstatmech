%\generatetitle{Langevin small approximation}
\label{chap:langevinSmallApprox}
%
The Langevin function is given by
\begin{equation}\label{eqn:langevinSmallApprox:20}
L(x) = \coth x - \inv{x}.
\end{equation}
%
According to \citep{pathriastatistical}, the small \(x\) approximation of this is
\begin{equation}\label{eqn:langevinSmallApprox:40}
\frac{x}{3} - \frac{x^3}{45}.
\end{equation}
%
Let's confirm this.
\begin{equation}\label{eqn:langevinSmallApprox:60}
\begin{aligned}
L(x)
&= \coth x - \inv{x} \\
&= \frac{\cosh x}{\sinh x} - \inv{x} \\
&= \lr{ \frac{\cosh x - (\sinh x)/x}{\sinh x} } \\
&=
\frac{
  \lr{1 + x^2/2! + x^4/4! + \cdots}
- \lr{1 + x^2/3! + x^4/5! + \cdots}
}{\sinh x} \\
&=
\inv{\sinh x}
\lr{
   x^2
   \lr{
   \inv{2!}
   -\inv{3!}
   }
   +
   x^4
   \lr{
   \inv{4!}
   -\inv{5!}
   }
   + \cdots
} \\
&=
\frac{
   x^2
   \lr{
   \inv{2!}
   -\inv{3!}
   }
   +
   x^4
   \lr{
   \inv{4!}
   -\inv{5!}
   }
   + \cdots
}
{
x + \frac{x^3}{3!} + \cdots
} \\
&=
\frac{
   x
   \lr{
   \inv{2!}
   -\inv{3!}
   }
   +
   x^3
   \lr{
   \inv{4!}
   -\inv{5!}
   }
   + \cdots
}
{
1 + \frac{x^2}{3!} + \cdots
} \\
&\approx
\lr{
   x
   \lr{
   \inv{2!}
   -\inv{3!}
   }
   +
   x^3
   \lr{
   \inv{4!}
   -\inv{5!}
   }
   + \cdots
}
\lr{
1 - \frac{x^2}{3!} - \frac{x^4}{5!} - \cdots
} \\
&=
x
\lr{
\inv{2!}
-\inv{3!}
}
+
x^3
\lr{
\lr{
\inv{4!}
-\inv{5!}
}
-
\inv{3!}
\lr{
\inv{2!}
-\inv{3!}
}
}
+ \cdots \\
&= x - \frac{x^3}{45} + \cdots
\end{aligned}
\end{equation}
