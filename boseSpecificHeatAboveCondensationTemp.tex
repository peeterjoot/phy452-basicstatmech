\makeoproblem{Boson \(\CV\) above condensation temp.}{pr:boseSpecificHeatAboveCondensationTemp:1}{\citep{pathriastatistical} \S 7.1.37}{
Equation 7.1.33 provides a relation for specific heat for a Bose gas
\begin{equation}\label{eqn:boseSpecificHeatAboveCondensationTemp:20}
\frac{\CV}{N \kB} =
\left(
\PD{T}{}
\lr{
\frac{3}{2} T
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
}
\right)_v.
\end{equation}
Fill in the details showing how this can be used to find
\begin{equation}\label{eqn:boseSpecificHeatAboveCondensationTemp:40}
\frac{\CV}{N \kB} =
\frac{15}{4}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
-
\frac{9}{4}
\frac{ g_{3/2}(z) }
{ g_{1/2}(z) }.
\end{equation}
} % makeoproblem
%
\makeanswer{pr:boseSpecificHeatAboveCondensationTemp:1}{
%
With
\begin{equation}\label{eqn:boseSpecificHeatAboveCondensationTemp:60}
\gBose{3/2}
= \frac{\lambda^3}{v}
=
\frac{h^3}{\lr{2 \pi m \kB T}^{3/2}},
\end{equation}
we have for constant \(v\)
\begin{dmath}\label{eqn:boseSpecificHeatAboveCondensationTemp:80}
\PDc{T}{g_{3/2}}{v}
=
-\frac{3}{2}
\frac{h^3}{\lr{2 \pi m \kB}^{3/2} T^{5/2}}
=
-\frac{3}{2 T} \gBose{3/2}.
\end{dmath}
%
From the series expansion
\begin{equation}\label{eqn:boseSpecificHeatAboveCondensationTemp:100}
\gBose{\nu} = \sum_{k = 1}^\infty \frac{z^k}{k^\nu},
\end{equation}
%
we have
\begin{dmath}\label{eqn:boseSpecificHeatAboveCondensationTemp:120}
z \PD{z}{} \gBose{\nu} =
z
\sum_{k = 1}^\infty k \frac{z^{k-1}}{k^\nu}
=
\sum_{k = 1}^\infty \frac{z^{k}}{k^{\nu-1}}
= \gBose{\nu-1}.
\end{dmath}
%
Taken together we have
\begin{dmath}\label{eqn:boseSpecificHeatAboveCondensationTemp:140}
\begin{aligned}
-\frac{3}{2 T} \gBose{3/2}
&=
\PDc{T}{g_{3/2}}{v} \\
&=
\PDc{T}{z}{v}
\PD{z}{} \gBose{3/2} \\
&=
\inv{z} \PDc{T}{z}{v}
z \PD{z}{} \gBose{3/2} \\
&=
\inv{z} \PDc{T}{z}{v}
\gBose{1/2},
\end{aligned}
\end{dmath}
or
\begin{dmath}\label{eqn:boseSpecificHeatAboveCondensationTemp:160}
\inv{z} \PDc{T}{z}{v} = -\frac{3}{2 T}
\frac{\gBose{3/2}}{
\gBose{1/2}}
.
\end{dmath}
%
We are now ready to evaluate the derivative and find the specific heat
\begin{dmath}\label{eqn:boseSpecificHeatAboveCondensationTemp:180}
\frac{\CV}{N \kB}
=
\left(
\PD{T}{}
\lr{
\frac{3}{2} T
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
}
\right)_v
=
\frac{3}{2}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
+
\frac{3 T}{2}
\PDc{T}{z}{v}
\PD{z}{}
\lr{
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
}
=
\frac{3}{2}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
-
\frac{9 T}{4}
\frac{\gBose{3/2}}{
\gBose{1/2}}
z
\PD{z}{}
\lr{
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
}
=
\frac{3}{2}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
-
\frac{9 }{4}
\frac{\gBose{3/2}}{
\gBose{1/2}}
\cancel{
\frac{ g_{3/2}(z) }
{ g_{3/2}(z) }
}
+
\frac{9 }{4}
\frac{\cancel{\gBose{3/2}}}{
\cancel{\gBose{1/2}}}
\frac{ g_{5/2}(z) \cancel{g_{1/2}(z)}}
{ \lr{g_{3/2}(z)}^{\cancel{2}} }
=
\frac{3}{2}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
-
\frac{9 }{4}
\frac{\gBose{3/2}}{
\gBose{1/2}}
+
\frac{9 }{4}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
=
\frac{15}{4}
\frac{ g_{5/2}(z) }
{ g_{3/2}(z) }
-
\frac{9 }{4}
\frac{\gBose{3/2}}{
\gBose{1/2}}.
\end{dmath}
%
This is the desired result.
} % makeanswer
